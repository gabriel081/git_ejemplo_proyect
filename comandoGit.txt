
--Iniciamos el proyecto en la carpeta determinada 
$git init

--Ver estado de los archivos
$git status  (ver estado de archivos)

--Agregar un archivo 
$git add style.css   

--Configurar mi correo
$ git config --global user.email "gab.arnaldo@gmail.com"    

--Configurar con mi nombre de usuario
$git config --global user.name "Gabriel"

--Hacer un commit dentro del archivo abierto poner mi primer commit y luego escape y :wq"
$git commit

--Ver cambios
$git log 

--Ver estado working tree clean  "noy hay cambios"
$git status

--Deshacer cambios
$ git checkout -- index.html     

--Comparar Archivos
$git diff index.html   

--Hacer commit desde una sola linea
$ git commit -m "He agrado un gitinore"  

--Verion alternativa
$ git branch login  

--cambiar de master a login
$git checkout login  

--git agrego todos los archivos de una sola vez
git add .
